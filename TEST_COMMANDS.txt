# 2) Every SQL statement must end with ';'.

## 1) Database and table setup
CREATE DATABASE testdb;
CREATE DATABASE analytics;
USE testdb;

CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR, score DECIMAL);
CREATE TABLE products (product_id INT PRIMARY KEY, product_name VARCHAR, price DECIMAL);
CREATE TABLE events (event_id INT PRIMARY KEY, event_type VARCHAR, timestamp INT);

SELECT * FROM users;
SELECT * FROM products;
SELECT * FROM events;

## 5) Column projection
SELECT id, name FROM users;
SELECT product_id, product_name FROM products;
SELECT event_id, event_type FROM events;
SELECT name, score FROM users WHERE score > 90.0;

## 6) Join operations
SELECT * FROM users INNER JOIN products ON users.id = products.product_id;
SELECT users.id, users.name, products.product_name FROM users INNER JOIN products ON users.id = products.product_id;
SELECT users.name, events.event_type FROM users INNER JOIN events ON users.id = events.event_id;
SELECT users.name, events.event_type FROM users INNER JOIN events ON users.id = events.event_id WHERE events.timestamp > 1010000;
